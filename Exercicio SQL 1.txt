CREATE DATABASE LIVRARIA;

USE LIVRARIA;

CREATE TABLE AUTORES (AUTORID INT PRIMARY KEY, NOME VARCHAR(255), NACIONALIDADE VARCHAR(255));

CREATE TABLE EDITORAS (EDITORAID INT PRIMARY KEY, NOME VARCHAR(255), LOCAL VARCHAR(99));

CREATE TABLE LIVROS (LIVROID INT PRIMARY KEY, TITULO VARCHAR(255), ANO INT, AUTORID INT, EDITORAID INT, FOREIGN KEY (AUTORID) REFERENCES AUTORES(AUTORID), FOREIGN KEY (EDITORAID) REFERENCES EDITORAS(EDITORAID));
